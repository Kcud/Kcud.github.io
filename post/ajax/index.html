要添加spring依赖

//Controller类
@Controller
@RequestMapping("/ajax")
public class AjaxController {
    //第一种方式，服务器要返回一个字符串，直接使用response
    @RequestMapping("/a1")
    public void ajax1(String name, HttpServletResponse response){
        try{
            if("admin".equals(name)){
                response.getWriter().print("true");
            }else{
                response.getWriter().print("false");
            }
        }catch (Exception e){
            e.printStackTrace();
        }
    }
    @RequestMapping("/a2")
    @ResponseBody
    public List a2(){
        List<User> list = new ArrayList<User>();
        User user = new User();
        user.setName("Flower");
        user.setSex("male");
        list.add(user);
        return list;
    }
}
